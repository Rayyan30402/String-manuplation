DECLARE Str1 : STRING
DECLARE Str2 : STRING
DECLARE Char1 : CHRACTER
DECLARE Char2 : CHRACTER
DECLARE nextChar : CHRACTER
DECLARE Counter : INTEGER
Str1 = ""
Str2 = ""
Char1 = ''
Char2 = ''
nextChar = ''
Counter = 0

OUTPUT "Enter string: "
INPUT Str1

OUTPUT "Enter character to remove: "
INPUT Char1

OUTPUT "Enter character to replace: "
INPUT Char2

FOR Counter = 1 to LENGTH(Str1)
    nextChar = MID(Str1, Counter, 1)
    IF nextChar = Char1
      THEN
        nextChar = Char2
    ENDIF
    Str2 = Str2 & nextChar
ENDFOR
OUTPUT "Final string = " & Str2
